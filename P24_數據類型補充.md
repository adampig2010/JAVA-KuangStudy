

```java
public class P24 {
    public static void main(String[] args) {

        int i = 10;
        int i2 = 010;   // 八進制 0
        int i3 = 0x11;  // 十六進制 0x  0~9 A~F 16

        System.out.println(i);
        System.out.println(i2);
        System.out.println(i3);

        // ==========================================
        //浮點數   有限  離散  捨入誤差    接近但不等於
        //double 最好完全使用浮點數進行比較

        float f = 0.1f; //0.1
        double d = 1.0/10; //0.1

        System.out.println(f==d);   //false

        float d1 = 233131323231313434f;
        float d2 = d1+1;

        System.out.println(d1==d2); //true

        // ==========================================

        char c1 = 'A';
        char c2 = '中';

        System.out.println(c1);
        System.out.println((int)c1);
        System.out.println(c2);
        System.out.println((int)c2);

        //所有字符本質還是數字
        //編碼 Unicode 表(a=97 / A=65) 2字節 0-65536 excel 2 16 65536
        //U0000  UFFFF

        char c3 = '\u0061';
        System.out.println(c3); //a

        // ==========================================

        //轉義字符
        // \t
        // \n
        System.out.println("Hello \t\n world");

        // ==========================================

        String sa = new String("hello world");
        String sb = new String("hello world");
        System.out.println(sa==sb); //false

        String sc = "hello world";
        String sd = "hello world";
        System.out.println(sc==sd); //true
        //對象 從內存分析

        // ==========================================
        // 布林值拓展
        boolean flag = true;
        if(flag == true){}; //新手
        if(flag){}  //老手

        ///Less is More! 代碼要精簡易讀
    }
}
```
